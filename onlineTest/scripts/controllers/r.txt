


































var express = require('express');
var router = express.Router();
var sharedObj = require('./shared');

//users registrations page(admin,school,student)
router.post('/user-reg', function(req, res) {
    var uid = req.body.uid;
    var pass = req.body.pass;
    var email = req.body.email;
    var phone = req.body.phone;
    var role = req.body.role;

    var con = sharedObj.getCon();
    //connect to the mysql database
    con.connect(function(e, r) {
        if (e) {
            res.send(e);
        }
    });
    //prepare query
    var query = "insert into users(uid,pass,email,phone,role) values('" + uid + "','" + pass + "','" + email + "'," + phone + ",'" + role + "')";

    //execute query
    con.query(query, function(error, response) {
        if (error) {
            res.send(error);
        } else {
            res.send(response);
        }
    })
});
